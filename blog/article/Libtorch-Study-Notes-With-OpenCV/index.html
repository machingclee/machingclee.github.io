How to deal with .npy file

There are several packages to load .npy file in C++. However, since have pytorch, additional package for .npy file is not necessary.

Save the Numpy Array as Tensor in Pytorch

We can load the array from .npy file and translate it into tensor easily by

1anchors = torch.tensor(np.load(anchors_npy_path), dtype=torch.float32)

then we save it by

1torch.save({"anchors": anchors}, "anchors.pt")

12void nnModule::load_parameters(std::string pt_pth)
13{
14    std::vector<char> f               = this->get_the_bytes(pt_pth);
15    c10::Dict<IValue, IValue> weights = torch::pickle_load(f).toGenericDict();
16
17    const torch::OrderedDict<std::string, at::Tensor>& model_params = this->named_parameters();
18    std::vector<std::string> param_names;
19    for (auto const& w : model_params)
20    {
21        param_names.push_back(w.key());
22    }
23
24    torch::NoGradGuard no_grad;
25    for (auto const& w : weights)
26    {
27        std::string name = w.key().toStringRef();
28        at::Tensor param = w.value().toTensor();
29
30        if (std::find(param_names.begin(), param_names.end(), name) != param_names.end())
31        {
32            auto target_model_param = model_params.find(name);
33
34            for (int i = 0; i < target_model_param->sizes().size(); i++)
35            {

37            }
38

40        }
41        else

44        };

Example

For example, in this repository my modules are all inherited publicly from nnModule defined above, therefore we can

When two files are exactly the same, we are confident that the pytorch weight can be applied to libtorch model as well.

Register Tensors form Pytorch to Libtorch

The following are equivalent:

1class Net(torch.nn.Module):
2  def __init__(self, N, M):
3    super(Net, self).__init__()
4    self.W = torch.nn.Parameter(torch.randn(N, M))
5    self.b = torch.nn.Parameter(torch.randn(M))
6
7  def forward(self, input):

1class Net : torch::nn::Module
2{
3    torch::Tensor W, b;
4public:
5    Net(int64_t N, int64_t M)
6    {
7        W = register_parameter("W", torch::randn({N, M}));
8        b = register_parameter("b", torch::randn(M));
9    }
10    torch::Tensor forward(torch::Tensor input)
11    {

Modules

The following are equivalent:

1class Net(torch.nn.Module):
2  def __init__(self, N, M):
3      super(Net, self).__init__()
4      # Registered as a submodule behind the scenes
5      self.linear = torch.nn.Linear(N, M)
6      self.another_bias = torch.nn.Parameter(torch.rand(M))
7
8  def forward(self, input):
9    return self.linear(input) + self.another_bias
1class Net : torch::nn::Module
2{
3    torch::nn::Linear linear;
4    torch::Tensor another_bias;
5public:
6    Net(int64_t N, int64_t M) : {
7        linear = register_module("linear", torch::nn::Linear(N, M));
8        another_bias = register_parameter("b", torch::randn(M));
9    }
10    torch::Tensor forward(torch::Tensor input) {
11        return linear->forward(input) + another_bias;
12    }
13};
Real Example in Practice
Rules
  • In libtorch all modules are created by producing a shared_ptr pointing to an nn::Module.
  • For example, auto net1 = nn::Conv2d(...) and auto net2 = nn::Relu(...) are both pointers.
  • nn::Sequential() only accepts smart pointers like nn::Sequential(net1, net2).

For custom module, we can create such a pointer-factory by TORCH_MODULE macro. The principles are

  • For module just for internal use, we simply leave it as an nn::Module object.
  • For module that is going to be exposed to user, we write NetImpl and use TORCH_MODULE(Net) to create a special pointer class Net.

2BlazeBlockImpl::BlazeBlockImpl(
3    int in_channels,
4    int out_channels,
5    int kernel_size,
6    int stride,
7    std::string act,
8    bool use_skip_proj) : in_channels(in_channels),
9                          out_channels(out_channels),
10                          kernel_size(kernel_size),
11                          stride(stride),
12                          act(act),
13                          use_skip_proj(use_skip_proj)
14{
15    channel_pad = out_channels - in_channels;
16    if (stride == 2)
17    {
18        max_pool = nn::MaxPool2d(nn::MaxPool2dOptions({ stride, stride }));
19        padding  = 0;
20    }
21    else
22    {
23        padding = (int)((kernel_size - 1) / 2);
24    }
25
26    nn::Sequential convs_ = nn::Sequential();
27    convs_->push_back(nn::Conv2d(nn::Conv2dOptions(in_channels, in_channels, kernel_size)
28                                      .stride(stride)
29                                      .padding(padding)
30                                      .groups(in_channels)
31                                      .bias(true)));
32    convs_->push_back(nn::Conv2d(nn::Conv2dOptions(in_channels, out_channels, 1)
33                                      .stride(1)
34                                      .padding(0)
35                                      .bias(true)));
36
37    convs = register_module("convs", convs_);
38
39    if (use_skip_proj)
40    {
41        skip_proj = register_module(
42            "skip_proj",
43            nn::Conv2d(nn::Conv2dOptions(in_channels, out_channels, 1)
44                            .stride(1)
45                            .padding(0)
46                            .bias(true)));
47    }
48    else
49    {
50        skip_proj = nullptr;
51    }
52
53    if (act == "relu")
54    {

56    }
57    else if ("prelu")
58    {
59        act_layer = register_module(
60            "act",

70torch::Tensor BlazeBlockImpl::forward(torch::Tensor x)
71{
72    torch::Tensor h;
73    if (stride == 2)
74    {
75        if (kernel_size == 3)
76        {
77            h = nn::functional::pad(x, nn::functional::PadFuncOptions({ 0, 2, 0, 2 }).value(0));
78        }
79        else
80        {
81            h = nn::functional::pad(x, nn::functional::PadFuncOptions({ 1, 2, 1, 2 }).value(0));
82        }
83        x = this->max_pool(x);
84    }
85    else
86    {
87        h = x;
88    }
89
90    if (skip_proj)
91    {
92        x = skip_proj->forward(x);
93    }
94    else if (channel_pad > 0)
95    {
96        x = nn::functional::pad(
97            x, n
98            n::functional::PadFuncOptions({ 0, 0, 0, 0, 0, channel_pad }).value(0)
99        );
100    }
101
102    torch::Tensor y = convs->forward(h) + x;
103    // y = reinterpret_cast<IHasForward*>(&act_layer)->forward(y);

105
106    if (auto act_layer_ptr = std::get_if<nn::ReLU>(&act_layer))
