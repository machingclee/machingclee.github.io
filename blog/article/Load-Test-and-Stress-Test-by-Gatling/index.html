We can use jsonPath $.token to extract this value and use it elsewhere.

1package videogamedb.scriptfundamentals
2import io.gatling.core.Predef._
3import io.gatling.http.Predef._
4
5class Authenticate extends Simulation {
6  val httpProtocol = http.baseUrl("https://videogamedb.uk/api")
7    .acceptHeader("application/json")
8    .contentTypeHeader("application/json")
9
10  def authenticate() = {
11    exec(http("Authenticate")
12      .post("/authenticate")
13      .body(StringBody("{\n  \"password\": \"admin\",\n  \"username\": \"admin\"\n}"))
14      .check(jsonPath("$.token").saveAs("jwtToken"))
15    )
16  }
17
18  def createNewGame() = {
19    exec(http("Create new game")
20      .post("/videogame")
21      .header("Authorization", "Bearer #{jwtToken}")

23        "{\n  \"category\": \"Platform\",\n  \"name\": \"Mario\",\n  \"rating\": \"Mature\",\n  \"releaseDate\": \"2012-05-04\",\n  \"reviewScore\": 85\n}"
24      ))
25    )
26  }
27
28  val scn = scenario("Authenticate")
29    .exec(authenticate())
30    .exec(createNewGame())

9class ComplexCustomFeeder extends Simulation{
10  val httpProtocol = http.baseUrl("https://videogamedb.uk/api")

12    .contentTypeHeader("application/json")
13
14  val idNumbers = (1 to 10).iterator
15  val rnd = new Random()
16  val now = LocalDate.now()
17  val pattern = DateTimeFormatter.ofPattern("yyyy-MM-dd")
18
19  def randomString(length:Int) ={
20    rnd.alphanumeric.filter(_.isLetter).take(length).mkString
21  }
22
23  def randomDate(startDate: LocalDate, random:Random):String = {
24    startDate.minusDays(random.nextInt(30)).format(pattern)
25  }
26
27  val customFeeder = Iterator.continually(Map(
28    "gameId" -> idNumbers.next(),
29    "name" -> ("Game-" + randomString(5)),
30    "releaseDate" -> randomDate(now, rnd),
31    "reviewScore" -> rnd.nextInt(100),
32    "category" ->  ("Category-" + randomString(6)),
33    "rating" ->  ("Rating-" + randomString(4)),
34  ))
35
36  def authenticate() = {
37    exec(http("Authenticate")
38      .post("/authenticate")
39      .body(StringBody("{\n  \"password\": \"admin\",\n  \"username\": \"admin\"\n}"))
40      .check(jsonPath("$.token").saveAs("jwtToken"))
41    )
42  }
43
44  def createNewGame()={
45    repeat(10){
46      feed(customFeeder)
47        .exec(http("Create new game - #{name}").
48          post("/videogame")
49          .header("authorization", "Bearer #{jwtToken}")
50          .body(ElFileBody("bodies/newGameTemplate.json")).asJson
51          .check(bodyString.saveAs("responseBody"))
