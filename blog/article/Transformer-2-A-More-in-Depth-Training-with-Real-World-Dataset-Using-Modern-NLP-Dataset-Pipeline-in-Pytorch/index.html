New Practice from torchtext 0.10.1 onwards

Old NLP material in pytorch usually import the following:

1# or from torchtext.legacy.data
2from torchtext.data import Field, Field, BucketIterator, TabularDataset

but these are completely removed from torchtext 0.10.1 onwards. It seems that the official pytorch suggests users creating their dataset via traditional classes:

1from torch.utils.data import Dataset, DataLoader
spacy package

  • We will rely on a package called spacy for tokenization. We will need this in our Corpus class, which is called implicitly in get_tokenizer imported from torchtext.data.utils.

  • We install the spacy and the desired lanauges by following the guide in https://spacy.io/usage.

Prepare Dataset Pipeline

Suppose that our dataset is a txt file in which every line is an English sentence paired with its Japanese translation, separated by \t:

1or the relative risk of drugs	ほぼ無関係です
2gail, are you drunk?	ゲイル 酔ってる?
3be careful.	注意しろ
4...

Let's first define our Corpus class that iterates the dataset to provide these sentences.

Note that get_src_sentence_iter and get_tgt_sentence_iter below are dependent on the structure of the dataset.

12class Corpus:
13    def __init__(
14            self,
15            src_lang="en_core_web_sm",
16            tgt_lang="ja_core_news_sm",
17            delimiter="\t",
18            src_vocab: Optional[Vocab] = None,
19            tgt_vocab: Optional[Vocab] = None
20        ):
21        # the lang keys are used in defining "field object"
22        # which is exactly the csv's header, the column name, the json key, etc.
23        self.delimiter=delimiter
24
25        self.src_tokenizer = get_tokenizer("spacy", language=src_lang)
26        self.tgt_tokenizer = get_tokenizer("spacy", language=tgt_lang)
27
28        src_counter = Counter()
29        tgt_counter =  Counter()
30
31        if src_vocab is not None and tgt_vocab is not None:
32            self.src_vocab = src_vocab
33            self.tgt_vocab = tgt_vocab
34        else:
35            for src_line in self.get_src_sentence_iter():
36                src_counter.update(self.src_tokenizer(src_line))
37            for tgt_line in self.get_tgt_sentence_iter():
38                tgt_counter.update(self.tgt_tokenizer(tgt_line))
39
40            # for label, line in
41            self.src_vocab = vocab(
42                src_counter,
43                min_freq=2,
44                specials=('<ukn>', '<pad>')
45            )
46            self.tgt_vocab = vocab(
47                tgt_counter,
48                min_freq=2,
49                specials=('<ukn>', '<sos>', '<eos>', '<pad>')
50            )
51
52    def get_src_sentence_iter(self):
53        with open(config.data_path, encoding="utf-8") as f:
54            for line in f:
55                src_line, _ = line.split(self.delimiter)
56                yield src_line
57
58    def get_tgt_sentence_iter(self):
59        with open(config.data_path, encoding="utf-8") as f:
60            for line in f:
61                _, tgt_line = line.split(self.delimiter)
62                yield tgt_line
63
64    def save_vocabs(self):
65        vocabs = {
66            "src": self.src_vocab,
67            "tgt": self.tgt_vocab
68        }
69        for lang, vocab in vocabs.items():
70            with open(f"{lang}.pickle", 'wb+') as handle:
71                pickle.dump(vocab, handle, protocol=pickle.HIGHEST_PROTOCOL)

Since creating vocabs involve accessing to all sentences in a dataset, it can take quite long and it is worth saving the vocabs, which include

  • index_to_word (obtained by Vocab.get_itos()) and
  • word_to_index (obtained by Vocab.get_stoi()) somewhere else so that we can reuse later without iterating the whole dataset again, that's why we have the save_vocabs method.

After saving the vocabs into pickle files, we can retrieve it by the class:

1class Vocabs:
2    src_vocab = None
3    tgt_vocab = None
4
5    def __init__(self, src_vocab_pickle_path, tgt_vocab_pickle_path):
6        self.src_vocab_pickle_path = src_vocab_pickle_path
7        self.tgt_vocab_pickle_path = tgt_vocab_pickle_path
8
9    def get_src_vocab(self) -> Vocab:
10        if Vocabs.src_vocab is None:
11            with open(self.src_vocab_pickle_path, 'rb') as handle:
12                Vocabs.src_vocab = pickle.load(handle)
13
14        return Vocabs.src_vocab
15
16    def get_tgt_vocab(self):
17        if Vocabs.tgt_vocab is None:
18            with open(self.tgt_vocab_pickle_path, 'rb') as handle:
19                Vocabs.tgt_vocab = pickle.load(handle)
20
21        return Vocabs.tgt_vocab

1class TransformerDataset(Dataset):
2    def __init__(self, corpus: Corpus):
3        self.corpus = corpus
4        self.src_sentences = list(self.corpus.get_src_sentence_iter())
5        self.tgt_sentences = list(self.corpus.get_tgt_sentence_iter())
6
7    def __getitem__(self, index):
8        src_text = self.src_sentences[index]
9        tgt_text = self.tgt_sentences[index]
10
11        src_stoi = self.corpus.src_vocab.get_stoi()
12        tgt_stoi =  self.corpus.tgt_vocab.get_stoi()
13
14        src_tokens= self.corpus.src_tokenizer(src_text)
15        tgt_tokens = self.corpus.tgt_tokenizer(tgt_text)
16
17        src_pad_len = config.src_max_len - len(src_tokens)
18        tgt_pad_len = config.tgt_max_len - len(tgt_tokens)
19
20        if src_pad_len > 0:
21            src_idxes = [src_stoi.get(token, src_stoi["<ukn>"]) for token in src_tokens] + [src_stoi["<pad>"]] * src_pad_len
22        else:
23            src_idxes = [src_stoi.get(token, src_stoi["<ukn>"]) for token in src_tokens[:config.src_max_len]]
24
25        if tgt_pad_len > 0:
26            tgt_idxes = [tgt_stoi['<sos>']] + \
27                        [tgt_stoi.get(token, src_stoi["<ukn>"]) for token in tgt_tokens] + \
28                        [tgt_stoi['<eos>']] + \
29                        [tgt_stoi["<pad>"]] * tgt_pad_len
30        else:
31            tgt_idxes = [tgt_stoi['<sos>']] + \
32                        [tgt_stoi.get(token, src_stoi["<ukn>"]) for token in tgt_tokens[:config.tgt_max_len]] + \
33                        [tgt_stoi['<eos>']]
34
35        return torch.as_tensor(src_idxes, device=device), torch.as_tensor(tgt_idxes, device=device)
36
37
38    def __len__(self):
39        return len(self.src_sentences)

According to our dataset output we change our training script as follows, we always feed our model by "indexed" version of our naive dataset.

3def train(
4    epochs=10,
5    use_saved_vocab=False,
6    learning_rate=1e-3
7):
8    if use_saved_vocab:

21    ).to(device)
22

27                             batch_size=config.batch_size,
28                             shuffle=True
29                             )
30    # when arrived this step, pickle file must have been saved

New Translator

Next we build our translator based on the vocab objects:

1class Translator():
2    def __init__(self, transformer: Transformer):
3        self.transformer = transformer
4
5    def translate_input_index(self, enc_input, src_start_index, tgt_word_index, tgt_index_word):
6        dec_input = torch.zeros(1, 0).type_as(enc_input)
7        terminated = False
8        next_tgt_word_index = src_start_index
9        word_count = 0
10        while not terminated:
11            dec_input = torch.cat(
12                [
13                    dec_input.detach(),
14                    torch.tensor([[next_tgt_word_index]],dtype=enc_input.dtype).to(device)
15                ],
16                -1
17            )
18            word_count += 1
19            dec_output_logits, _, _, _= self.transformer(enc_input, dec_input)
20            next_tgt_word_index = torch.argmax(dec_output_logits[-1])
21
22            if next_tgt_word_index == tgt_word_index["<eos>"] or word_count == config.tgt_max_len + 1:
23                terminated = True
24
25        # remove batch, remove <sos>
26        return dec_input.squeeze(0)[1:]
